Массив в JavaScript

  • Объект похожий на список с некоторыми методами и свойствами
  • Ни размер, ни тип не фиксированы 
    
var emptyArray = [];
var stuff = [1, null, { well: 'hey there' }, [1, 2, 3]];

var a1 = new Array();                // a1.toString() -> ''
var a2 = new Array(3);               // a2.toString() -> ',,'
var a3 = new Array(1, 2, 3);         // a3.toString() -> '1,2,3'
  • Длина массива не является верхней границей

Область видимости функции

  x = 'inside h';
}

h();

Variable hoisting


var x = 42;

var f = function() {
  if (x === 42) {
    console.log(x);
  } else {
    var x = 50;
    console.log(x);
  }
}

f();

Вложенные области видимости


var factorial = function(n) {
  var factorialIter = function(i, acc) {
    var result;
    if (i < 2) {
      result = acc;
    } else {
      result = factorialIter(i - 1, acc * i);
    }
    return result;
  }

  return factorialIter(n, 1);
}

factorial(4)                                  // 24
factorialIter(6, 1);                          // error

Блочная vs Функциональная области видимости

В текущей версии присутствует только функциональная область видимости


function f(a) {
  if (a < 0) {
    var i = 3;
  }

  console.log(i);                             // 3
}

f(-1);

Behold the future (ECMAScript 6)


function f(a) {
   if (a < 0) {
      let i = 3;
   }

}

f(-1);

Необязательные аргументы


function log(text, level) {
  if (level == undefined)
    level = 'debug';

  console.log(level + ': ', text);
}

log('called with date');              // debug: called with date
log('called with string', 'error');   // error: called with string
log('called with numeric', 'info');   // info: called with numeric

  console.log(value);

function wrapValue(n) {
  var localVariable = n;
  return function() { return localVariable; };
}

var wrap1 = wrapValue(1);
var wrap2 = wrapValue(2);

console.log(wrap1());                         // 1
console.log(wrap2());                         // 2

Рекурсия


function factorial(n) {
  if (n < 2) {
    return 1;
  } else {
    return n * factorial(n - 1);
  }
}

factorial(4);                                 // 24
					

function findSolution(target) {
  function find(start, history) {
    if (start == target)
      return history;
    else if (start > target)
      return null;
    else
      return find(start + 5, "(" + history + " + 5)") ||
             find(start * 3, "(" + history + " * 3)");
  }
  return find(1, "1");
}

console.log(findSolution(24));
